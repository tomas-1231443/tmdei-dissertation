\chapter{Research Questions}

        \begin{itemize}

            \item Quais são os principais desafios envolvidos na implementação de um bot de inteligência artificial integrado a sistemas de gerenciamento de informações e eventos de segurança (SIEM) e de que forma esses desafios podem ser mitigados?
        \end{itemize}
    ]
